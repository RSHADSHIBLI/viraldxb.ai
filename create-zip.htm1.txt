    <script type="text/template" id="js_code">
const hooks = {
    'توعوي': ['هل تعلم أن TOPIC يغير حياتك؟', 'سر TOPIC الذي لا يعرفه 99%'],
    'ترفيهي': ['تحدي TOPIC 😱', 'جربت TOPIC لمدة يوم كامل'],
    'تسويقي': ['كيف تربح من TOPIC؟', 'استراتيجية TOPIC للمبتدئين'],
    'تحفيزي': ['من الصفر لـ TOPIC', 'قصتي الملهمة مع TOPIC']
};

function generate() {
    const type = document.getElementById('type').value;
    const topic = document.getElementById('topic').value || 'هذا المجال';
    
    const list = hooks[type];
    let html = '';
    list.forEach(h => {
        html += '<div class="hook">' + h.replace('TOPIC', topic) + '</div>';
    });
    
    document.getElementById('result').innerHTML = html;
    
    // حفظ تلقائي
    localStorage.setItem('lastHooks', html);
}

// استعادة آخر نتائج
window.onload = function() {
    const saved = localStorage.getItem('lastHooks');
    if (saved) document.getElementById('result').innerHTML = saved;
};
